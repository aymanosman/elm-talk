\subsection{Anonymous Functions}

\begin{frame}[fragile]
  \frametitle{\insertsection: Anonymous Functions}
\begin{lstlisting}
      -- JavaScript
      var f = function(x) { return x + 1; };
      var g = (x) => x + 1;

      -- Elm
      f = \x -> x + 1
\end{lstlisting}

  Math

  f = \lambda x. x + 1

\end{frame}

\subsection{Currying}

\begin{frame}[fragile]
  \frametitle{\insertsection: Currying}

    \begin{lstlisting}
      add x y = x + y
      -- Is equivalent to
      add = \ x -> (\y -> x + y)

      -- So the function `addThree` can be defined as
      addThree = add 3

      addThree 4 == 7 -- True
    \end{lstlisting}


    \pause

    You can think of it as:

    \begin{quote}
      if you give a function too few arguments, it will give you back a function
      that expects the rest of the arguments before giving you the answer.
    \end{quote}
\end{frame}
